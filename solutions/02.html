Solution: letter censor

function letterCensor(letter, word) {
    let keepers = [];

    for (let i = 0; i < word.length; i++) {
        if (word[i] !== letter) {
        keepers.push(word[i]);
        }
    }

    return keepers.join('');
}


// Test(s)
letterCensor('a', 'not again avery!'); // returns 'not gin very!'
