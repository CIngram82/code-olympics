Solution: Mind the gap

function gap(nums) {
    let max = nums[0];
    let min = nums[0];

    for (let i = 0; i < nums.length; i++) {
        if (nums[i] > max) {
            max = nums[i];
        } else if (nums[i] < min) {
            min = nums[i];
        }
    }

    return max - min;
}

gap([1, 10, 6, 8, 2])   // return 9
gap([1, 2, 3])          // return 2
