function palindrome(words) {
    words = words.toLowerCase().split(' ').join('');

    for (let i = 0; i < words.length; i++) {
        let end = words.length - (i + 1);

        if (words[i] !== words[end]) {
            return false;
        }
    }

    return true;
}

palindrome('A Santa At NASA');
